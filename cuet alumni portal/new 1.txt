<script>
function alumnicatOnchange(){
    var visa_cat = document.getElementById('searchEmployee').value 
    if(visa_cat == "name")
	    {
        document.getElementById("searchname").style.display="block";
		document.getElementById("searchroll").style.display="none";
		document.getElementById("searchcountry").style.display="none";
		document.getElementById("searchjob").style.display="none";
		document.getElementById("search").style.display="block";
		}
	else if(visa_cat == "id")
	    {
        document.getElementById("searchroll").style.display="block";
		 document.getElementById("searchname").style.display="none";
		 document.getElementById("searchjob").style.display="none";
		document.getElementById("searchcountry").style.display="none";
		document.getElementById("search").style.display="block";
		}
	else if(visa_cat == "country")
	    {
		document.getElementById("searchname").style.display="none";
		document.getElementById("searchroll").style.display="none";
		document.getElementById("searchjob").style.display="none";
        document.getElementById("searchcountry").style.display="block";
		document.getElementById("search").style.display="block";
		}
	else if(visa_cat == "company")
	    {
		document.getElementById("searchname").style.display="none";
		document.getElementById("searchroll").style.display="none";
		document.getElementById("searchjob").style.display="block";
        document.getElementById("searchcountry").style.display="none";
		document.getElementById("search").style.display="block";
		}		
    else {
        document.getElementById("searchname").style.display="none";
		document.getElementById("searchroll").style.display="none";
		document.getElementById("searchcountry").style.display="none";
		document.getElementById("search").style.display="none";
			}
}
 </script>
